-- 1. Helper function to calculate experience threshold (same as before)
-- This mirrors the logic from your profile.js (BASE_XP=100, INCREMENT_XP=25)
CREATE OR REPLACE FUNCTION get_total_exp_threshold(target_level INTEGER)
RETURNS INTEGER AS $$
DECLARE
    total_exp INTEGER := 0;
    level_iter INTEGER;
    base_xp INTEGER := 100;
    increment_xp INTEGER := 25;
BEGIN
    IF target_level <= 1 THEN
        RETURN 0;
    END IF;

    FOR level_iter IN 1..(target_level - 1) LOOP
        total_exp := total_exp + (base_xp + (increment_xp * (level_iter - 1)));
    END LOOP;
    RETURN total_exp;
END;
$$ LANGUAGE plpgsql IMMUTABLE;

-- 2. Trigger function to handle level-up logic (corrected internal checks)
CREATE OR REPLACE FUNCTION public.handle_level_up()
RETURNS TRIGGER AS $$
BEGIN
    -- This function is called for every INSERT or UPDATE on profiles (if pg_trigger_depth = 0).
    -- We first check if it's an INSERT or if 'experience' or 'level' actually changed during an UPDATE.
    IF (TG_OP = 'INSERT') OR 
       (TG_OP = 'UPDATE' AND (NEW.experience IS DISTINCT FROM OLD.experience OR NEW.level IS DISTINCT FROM OLD.level)) THEN
        
        -- Loop to handle multiple level-ups if sufficient XP is gained.
        -- For INSERTs, NEW.level will typically be the default (e.g., 1).
        -- For UPDATEs, NEW.level is the value before this trigger function might modify it further in the loop.
        WHILE NEW.experience >= get_total_exp_threshold(NEW.level + 1) LOOP
            NEW.level := NEW.level + 1;
            
            -- Optional: Log the level-up event into the activities table
            -- Ensure your 'activities' table structure and permissions allow this.
            BEGIN
                INSERT INTO public.activities (user_id, type, title, description, details, icon)
                VALUES (
                    NEW.id, 
                    'level_up', 
                    'Dosažena nová úroveň!', 
                    'Gratulujeme k dosažení úrovně ' || NEW.level || '!', 
                    jsonb_build_object('new_level', NEW.level, 'total_experience', NEW.experience),
                    'fa-angle-double-up' -- Example icon from your CSS
                );
            EXCEPTION WHEN OTHERS THEN
                RAISE WARNING 'Failed to insert level up activity for user % (level %): %', NEW.id, NEW.level, SQLERRM;
            END;
        END LOOP;
    END IF;
    
    RETURN NEW; -- Return the (potentially modified) row to be inserted/updated
END;
$$ LANGUAGE plpgsql SECURITY DEFINER; -- Use SECURITY DEFINER if INSERT into activities requires elevated privileges

CREATE OR REPLACE FUNCTION public.update_user_badges_count()
RETURNS TRIGGER AS $$
DECLARE
    v_user_id UUID;
    v_badges_count INTEGER;
BEGIN
    -- Определяем user_id в зависимости от операции (INSERT, DELETE, UPDATE)
    IF (TG_OP = 'DELETE') THEN
        v_user_id := OLD.user_id;
    ELSE -- INSERT or UPDATE
        v_user_id := NEW.user_id;
    END IF;

    -- Подсчитываем количество отзнаков для данного пользователя
    SELECT count(*)
    INTO v_badges_count
    FROM public.user_badges
    WHERE user_id = v_user_id;

    -- Обновляем столбец badges_count в таблице profiles
    UPDATE public.profiles
    SET badges_count = v_badges_count,
        updated_at = now() -- Также обновляем updated_at
    WHERE id = v_user_id;

    RAISE LOG '[TRIGGER update_user_badges_count] User ID: %, New Badges Count: %', v_user_id, v_badges_count;

    -- Возвращаем OLD для DELETE или NEW для INSERT/UPDATE
    IF (TG_OP = 'DELETE') THEN
        RETURN OLD;
    ELSE
        RETURN NEW;
    END IF;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

COMMENT ON FUNCTION public.update_user_badges_count() IS 'Updates the badges_count in the profiles table based on the user_badges table after an insert, update, or delete operation on user_badges.';

CREATE OR REPLACE FUNCTION get_user_topic_progress_summary(p_user_id uuid)
RETURNS TABLE (
    topic_name TEXT,
    progress_percentage DOUBLE PRECISION,
    last_studied_at TIMESTAMPTZ,
    topic_icon TEXT  -- Это имя столбца, которое ожидает ваш JavaScript (main.js)
)
AS $$
BEGIN
    RETURN QUERY
    WITH user_topic_activities AS (
        SELECT
            pa.topic_id,
            et.name AS calculated_topic_name,
            et.icon_class AS calculated_topic_icon, -- Используем ваш столбец 'icon_class'

            public.plan_activities pa

            public.exam_topics et ON pa.topic_id = et.id
        WHERE
            pa.user_id = p_user_id
        GROUP BY
            pa.topic_id, et.name, et.icon_class -- Группируем также по icon_class
    )
    SELECT
        uta.calculated_topic_name,
        CASE
            WHEN uta.total_activities > 0 THEN ROUND((uta.completed_activities::DECIMAL / uta.total_activities) * 100.0, 2)
            ELSE 0.0
        END AS progress_percentage,
        uta.latest_activity_update AS last_studied_at,
        uta.calculated_topic_icon   -- Возвращаем значение из et.icon_class
    FROM
        user_topic_activities uta
    ORDER BY
        uta.latest_activity_update DESC NULLS LAST, uta.calculated_topic_name;
END;
$$ LANGUAGE plpgsql
